120  Eigen::Transform<Scalar, 3, Eigen::Isometry> get_foot_pose(const LimbID& limb) const {
121  // Assign the value based on the foot planted
122  Eigen::Transform<Scalar, 3, Eigen::Isometry> Htl =
123  left_foot_is_planted ? get_torso_pose().inverse() : get_torso_pose().inverse() * get_swing_foot_pose();

125  left_foot_is_planted ? get_torso_pose().inverse() * get_swing_foot_pose() : get_torso_pose().inverse();
126 
127  // Return the desired pose of the specified foot
128  if (limb == LimbID::LEFT_LEG)
129  return Htl;
130  if (limb == LimbID::RIGHT_LEG)
131  return Htr;

143  /**
144  * @brief Reset walk engine time and initial poses.
145  */
146  void reset() {
147  // Initialize swing foot pose
148  Hps_start.translation() = Eigen::Matrix<Scalar, 3, 1>(0.0, get_foot_width_offset(), 0.0);
149  Hps_start.linear().setIdentity();
150 
151  // Initialize torso pose
152  Hpt_start.translation() = torso_sway_offset + torso_position_offset;
153  Hpt_start.linear() =
154  Eigen::AngleAxis<Scalar>(torso_pitch, Eigen::Matrix<Scalar, 3, 1>::UnitY()).toRotationMatrix();
155 
156  // Start time at end of step period to avoid taking a step when starting.
157  t = step_period;
158 
159  // Generate trajectories for swing foot and torso with zero velocity target
160  generate_walking_trajectories(Eigen::Matrix<Scalar, 3, 1>::Zero());
161 
162  // Set engine state to stopped
163  engine_state = WalkState::State::STOPPED;
164  }

166  /**
167  * @brief Run an update of the walk engine, updating the time and engine state and trajectories.
168  * @param dt Time step.
169  * @param velocity_target Requested velocity target (dx, dy, dtheta).
170  * @return Engine state.
171  */
172  WalkState::State update(const Scalar& dt, const Eigen::Matrix<Scalar, 3, 1>& velocity_target) {
173  if (velocity_target.isZero() && t < step_period) {
174  // Requested velocity target is zero and we haven't finished taking a step, continue stopping
175  engine_state = WalkState::State::STOPPING;
176  }
177  else if (velocity_target.isZero() && t >= step_period) {
178  // Requested velocity target is zero and we have finished taking a step, remain stopped
179  engine_state = WalkState::State::STOPPED;
180  }
181  else {
182  // Requested velocity target is non-zero, walk
183  engine_state = WalkState::State::WALKING;
184  }
185 
186  switch (engine_state.value) {
187  case WalkState::State::WALKING:
188  update_time(dt);
189  // If we are at the end of the step, switch the planted foot and reset time
190  if (t >= step_period) {
191  switch_planted_foot();
192  }
193  break;
194  case WalkState::State::STOPPING:
195  update_time(dt);
196  // We do not switch the planted foot here because we want to transition to the standing state
197  break;
198  case WalkState::State::STOPPED:
199  // We do not update the time here because we want to remain in the standing state
200  break;
201  default: NUClear::log<NUClear::WARN>("Unknown state"); break;
202  }
203 
204  // Generate the torso and swing foot trajectories to reach the next foot step location
205  generate_walking_trajectories(velocity_target);
206 
207  return engine_state;
208  }

289  /**
290  * @brief Generate walking trajectories for the torso and swing foot.
291  * @param velocity_target Requested velocity target (dx, dy, dtheta).
292  */
293  void generate_walking_trajectories(const Eigen::Matrix<Scalar, 3, 1>& velocity_target) {
294  // Compute the next step placement in the planted foot frame based on the requested velocity target
295  Eigen::Matrix<Scalar, 3, 1> step = Eigen::Matrix<Scalar, 3, 1>::Zero();
296  step.x() = std::max(std::min(velocity_target.x() * step_period, step_limits.x()), -step_limits.x());
297  step.y() = std::max(std::min(velocity_target.y() * step_period, step_limits.y()), -step_limits.y());
298  step.z() = std::max(std::min(velocity_target.z() * step_period, step_limits.z()), -step_limits.z());
299 
300  // Create a waypoint variable to use for all waypoints
301  Waypoint<Scalar> waypoint;
302 
303  // ******************************** Swing Foot Trajectory ********************************
304  swingfoot_trajectory.clear();
305 
306  // Start waypoint: Start from the current swing foot position at time = 0
307  waypoint.time_point = 0.0;
308  waypoint.position = Hps_start.translation();
309  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
310  waypoint.orientation = MatrixToEulerIntrinsic(Hps_start.rotation());
311  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
312  swingfoot_trajectory.add_waypoint(waypoint);
313 
314  // Middle waypoint: Raise foot to step height at time = step_apex_ratio * step_period
315  waypoint.time_point = step_apex_ratio * step_period;
316  waypoint.position = Eigen::Matrix<Scalar, 3, 1>(0, get_foot_width_offset(), step_height);
317  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>(velocity_target.x(), velocity_target.y(), 0);
318  waypoint.orientation =
319  Eigen::Matrix<Scalar, 3, 1>(0.0, 0.0, velocity_target.z() * step_apex_ratio * step_period);
320  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>(0.0, 0.0, velocity_target.z());
321  swingfoot_trajectory.add_waypoint(waypoint);
322 
323  // End waypoint: End at next foot placement on ground at time = step_period
324  waypoint.time_point = step_period;
325  waypoint.position = Eigen::Matrix<Scalar, 3, 1>(step.x(), get_foot_width_offset() + step.y(), 0);
326  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
327  waypoint.orientation = Eigen::Matrix<Scalar, 3, 1>(0, 0, step.z());
328  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
329  swingfoot_trajectory.add_waypoint(waypoint);
330 
331  // ******************************** Torso Trajectory ********************************
332  torso_trajectory.clear();
333 
334  // Start waypoint: Start from the current torso position at time = 0
335  waypoint.time_point = 0.0;
336  waypoint.position = Hpt_start.translation();
337  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
338  waypoint.orientation = MatrixToEulerIntrinsic(Hpt_start.rotation());
339  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
340  torso_trajectory.add_waypoint(waypoint);
341 
342  // Middle waypoint: Shift torso to the torso_sway_offset at time = torso_sway_ratio * step_period
343  waypoint.time_point = torso_sway_ratio * step_period;
344  Scalar torso_offset_y = left_foot_is_planted ? -torso_sway_offset.y() : torso_sway_offset.y();
345  waypoint.position =
346  Eigen::Matrix<Scalar, 3, 1>(torso_sway_offset.x(), torso_offset_y, torso_height + torso_sway_offset.z())
347  + torso_position_offset;
348  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>(velocity_target.x(), velocity_target.y(), 0);
349  waypoint.orientation =
350  Eigen::Matrix<Scalar, 3, 1>(0.0, torso_pitch, velocity_target.z() * torso_sway_ratio * step_period);
351  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>(0.0, 0.0, velocity_target.z());
352  torso_trajectory.add_waypoint(waypoint);
353 
354  // End waypoint: Shift torso back to the final torso position at time = step_period
355  waypoint.time_point = step_period;
356  waypoint.position =
357  Eigen::Matrix<Scalar, 3, 1>(torso_final_position_ratio.x() * step.x(),
358  get_foot_width_offset() / 2 + torso_final_position_ratio.y() * step.y(),
359  torso_height)
360  + torso_position_offset;
361  waypoint.velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
362  waypoint.angular_velocity = Eigen::Matrix<Scalar, 3, 1>::Zero();
363  waypoint.orientation =
364  Eigen::Matrix<Scalar, 3, 1>(0.0, torso_pitch, torso_final_position_ratio.z() * step.z());
365  torso_trajectory.add_waypoint(waypoint);
366  }

368  /**
369  * @brief Switch planted foot.
370  * @details This function switches the planted foot and updates the start torso and start swing foot poses.
371  */
372  void switch_planted_foot() {
373  // Transform planted foot to swing foot frame at end of step
374  Hps_start = get_swing_foot_pose(step_period).inverse();
375 
376  // Transform torso to end torso frame at end of step, in the new planted foot frame
377  Hpt_start = Hps_start * get_torso_pose(step_period);
378 
379  // Switch planted foot indicator
380  left_foot_is_planted = !left_foot_is_planted;
381 
382  // Reset time
383  t = 0;
384  }

386  /**
387  * @brief Update the time.
388  * @param dt Time step.
389  */
390  void update_time(const Scalar& dt) {
391  // Check for negative time step
392  if (dt <= 0.0f) {
393  NUClear::log<NUClear::WARN>("dt <= 0.0f");
394  return;
395  }
396 
397  // Check for too long dt
398  if (dt > step_period) {
399  NUClear::log<NUClear::WARN>("dt > params.step_period");
400  return;
401  }
402 
403  // Update the phase
404  t += dt;
405 
406  // Clamp time to step period
407  if (t >= step_period) {
408  t = step_period;
409  }
410  }
