47  struct FootDownMethod {
48  enum Value { UNKNOWN = 0, Z_HEIGHT = 1, LOAD = 2, FSR = 3 };
49  Value value = Value::UNKNOWN;
50 
51  // Constructors
52  FootDownMethod() = default;
53  FootDownMethod(int const& v) : value(static_cast<Value>(v)) {}
54  FootDownMethod(Value const& v) : value(v) {}
55  FootDownMethod(std::string const& str) {
56  // clang-format off
57  if (str == "Z_HEIGHT") { value = Value::Z_HEIGHT; }
58  else if (str == "LOAD") { value = Value::LOAD; }
59  else if (str == "FSR") { value = Value::FSR; }
60  else {
61  value = Value::UNKNOWN;
62  throw std::runtime_error("String " + str + " did not match any enum for ServoID");
63  }
64  // clang-format on
65  }
66 
67  // Conversions
68  [[nodiscard]] operator Value() const {
69  return value;
70  }
71  [[nodiscard]] operator std::string() const {
72  switch (value) {
73  case Value::Z_HEIGHT: return "Z_HEIGHT";
74  case Value::LOAD: return "VIRTUAL";
75  case Value::FSR: return "FSR";
76  default: throw std::runtime_error("enum Method's value is corrupt, unknown value stored");
77  }
78  }

129  struct FootDown {

131  FootDown(const FootDownMethod& method, const std::map<FootDownMethod, float>& thresholds) {
132  set_method(method, thresholds);
133  }
134  void set_method(const FootDownMethod& method, const std::map<FootDownMethod, float>& thresholds) {
135  if (thresholds.count(method) == 0) {
136  throw std::runtime_error(fmt::format("Invalid foot down method '{}'", std::string(method)));
137  }
138  current_method = method;
139  certainty_thresholds = thresholds;
140  }
141  [[nodiscard]] float threshold() const {
142  return certainty_thresholds.at(current_method);
143  }
144  [[nodiscard]] FootDownMethod method() const {
145  return current_method;
146  }
147  FootDownMethod current_method = FootDownMethod::Z_HEIGHT;
148  std::map<FootDownMethod, float> certainty_thresholds = {
149  {FootDownMethod::Z_HEIGHT, 0.01f},
150  {FootDownMethod::LOAD, 0.05f},
151  {FootDownMethod::FSR, 60.0f},
152  };
153  } footDown;
