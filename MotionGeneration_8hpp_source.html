116  /**
117  * @brief Get the left or right foot pose at the given time in the torso {t} frame.
118  * @param t Time.
119  * @param limb Limb ID of foot to get pose of.
120  * @return Swing foot pose at time t.
121  */
122  Eigen::Transform<Scalar, 3, Eigen::Isometry> get_foot_pose(const LimbID& limb) const {
123  // Assign the value based on the foot planted
124  Eigen::Transform<Scalar, 3, Eigen::Isometry> Htl =
125  left_foot_is_planted ? get_torso_pose().inverse() : get_torso_pose().inverse() * get_swing_foot_pose();
126  Eigen::Transform<Scalar, 3, Eigen::Isometry> Htr =
127  left_foot_is_planted ? get_torso_pose().inverse() * get_swing_foot_pose() : get_torso_pose().inverse();
128 
129  // Return the desired pose of the specified foot
130  if (limb == LimbID::LEFT_LEG)
131  return Htl;
132  if (limb == LimbID::RIGHT_LEG)
133  return Htr;
134  throw std::runtime_error("Invalid Limb ID");
135  }

145  /**
146  * @brief Reset walk engine.
147  */
148  void reset() {
149  // Initialize swing foot pose
150  Hps_start.translation() = Eigen::Matrix<Scalar, 3, 1>(0.0, get_foot_width_offset(), 0.0);
151  Hps_start.linear().setIdentity();
152 
153  // Initialize torso pose
154  Hpt_start.translation() = Eigen::Matrix<Scalar, 3, 1>(0.0, get_foot_width_offset() / 2, torso_height);
155  Hpt_start.linear() =
156  Eigen::AngleAxis<Scalar>(torso_pitch, Eigen::Matrix<Scalar, 3, 1>::UnitY()).toRotationMatrix();
157 
158  // Initialize swing foot and torso trajectories
159  generate_swingfoot_trajectory(Eigen::Matrix<Scalar, 3, 1>::Zero());
160  generate_torso_trajectory(Eigen::Matrix<Scalar, 3, 1>::Zero());
161 
162  // Start time at end of step period to avoid taking a step when starting.
163  t = step_period;
164 
165  // Set engine state to stopped
166  engine_state = WalkState::State::STOPPED;
167  }
168 
169  /**
170  * @brief Run an update of the walk engine, updating the time and engine state and trajectories.
171  * @param dt Time step.
172  * @param velocity_target Requested velocity target (dx, dy, dtheta).
173  * @return Engine state.
174  */
175  WalkState::State update(const Scalar& dt, const Eigen::Matrix<Scalar, 3, 1>& velocity_target) {
176  if (velocity_target.isZero() && t < step_period) {
177  // Requested velocity target is zero and we haven't finished taking a step, continue stopping
178  engine_state = WalkState::State::STOPPING;
179  }
180  else if (velocity_target.isZero() && t >= step_period) {
181  // Requested velocity target is zero and we have finished taking a step, remain stopped
182  engine_state = WalkState::State::STOPPED;
183  }
184  else {
185  // Requested velocity target is non-zero, walk
186  engine_state = WalkState::State::WALKING;
187  }
188 
189  switch (engine_state.value) {
190  case WalkState::State::WALKING:
191  update_time(dt);
192  // If we are at the end of the step, switch the planted foot and reset time
193  if (t >= step_period) {
194  switch_planted_foot();
195  }
196  break;
197  case WalkState::State::STOPPING:
198  update_time(dt);
199  // We do not switch the planted foot here because we want to transition to the standing state
200  break;
201  case WalkState::State::STOPPED:
202  // We do not update the time here because we want to remain in the standing state
203  break;
204  default: NUClear::log<NUClear::WARN>("Unknown state"); break;
205  }
206 
207  // Generate the torso and swing foot trajectories
208  generate_swingfoot_trajectory(velocity_target);
209  generate_torso_trajectory(velocity_target);
210 
211  return engine_state;
212  }

288  /**
289  * @brief Generate swing foot trajectory.
290  * @param velocity_target Requested velocity target (dx, dy, dtheta).
291  * @param Hps_end Next foot placement.
292  * @return Trajectory of swing foot to follow to reach next foot placement.
293  */
294  void generate_swingfoot_trajectory(const Eigen::Matrix<Scalar, 3, 1>& velocity_target) {
295  // Clear current trajectory
296  swingfoot_trajectory.clear();
297 
298  // Compute next foot placement and clamp with step limits
299  Eigen::Matrix<Scalar, 3, 1> step_placement;
300  step_placement.x() =
301  std::max(std::min(velocity_target.x() * step_period, step_limits.x()), -step_limits.x());
302  step_placement.y() =
303  std::max(std::min(velocity_target.y() * step_period, step_limits.y()), -step_limits.y());
304  step_placement.z() =
305  std::max(std::min(velocity_target.z() * step_period, step_limits.z()), -step_limits.z());
306 
307  // X position trajectory
308  swingfoot_trajectory.add_waypoint(X, 0, Hps_start.translation().x(), 0);
309  swingfoot_trajectory.add_waypoint(X, step_apex_ratio * step_period, 0, velocity_target.x());
310  swingfoot_trajectory.add_waypoint(X, step_period, step_placement.x(), 0);
311 
312  // Y position trajectory
313  swingfoot_trajectory.add_waypoint(Y, 0, Hps_start.translation().y(), 0);
314  swingfoot_trajectory.add_waypoint(Y,
315  step_apex_ratio * step_period,
316  get_foot_width_offset(),
317  velocity_target.y());
318  swingfoot_trajectory.add_waypoint(Y, step_period, get_foot_width_offset() + step_placement.y(), 0);
319 
320  // Z position trajectory
321  swingfoot_trajectory.add_waypoint(Z, 0, Hps_start.translation().z(), 0);
322  swingfoot_trajectory.add_waypoint(Z, step_apex_ratio * step_period, step_height, 0);
323  swingfoot_trajectory.add_waypoint(Z, step_period, 0, 0);
324 
325  // Roll trajectory
326  swingfoot_trajectory.add_waypoint(ROLL, 0, 0, 0);
327  swingfoot_trajectory.add_waypoint(ROLL, step_apex_ratio * step_period, 0, 0);
328  swingfoot_trajectory.add_waypoint(ROLL, step_period, 0, 0);
329 
330  // Pitch trajectory
331  swingfoot_trajectory.add_waypoint(PITCH, 0, 0, 0);
332  swingfoot_trajectory.add_waypoint(PITCH, step_apex_ratio * step_period, 0, 0);
333  swingfoot_trajectory.add_waypoint(PITCH, step_period, 0, 0);
334 
335  // Yaw trajectory
336  swingfoot_trajectory.add_waypoint(YAW, 0, 0, 0);
337  swingfoot_trajectory.add_waypoint(YAW,
338  step_apex_ratio * step_period,
339  velocity_target.z() * step_apex_ratio * step_period,
340  0);
341  swingfoot_trajectory.add_waypoint(YAW, step_period, step_placement.z(), 0);
342  }

344  /**
345  * @brief Generate torso trajectory.
346  * @param velocity_target Requested velocity target (dx, dy, dtheta).
347  * @param Hps_end Next foot placement.
348  * @return Trajectory of torso to follow to reach next torso placement.
349  */
350  void generate_torso_trajectory(const Eigen::Matrix<Scalar, 3, 1>& velocity_target) {
351  // Clear current trajectory
352  torso_trajectory.clear();
353 
354  // X position trajectory
355  torso_trajectory.add_waypoint(X, 0, Hpt_start.translation().x(), 0);
356  torso_trajectory.add_waypoint(X,
357  torso_sway_ratio * step_period,
358  torso_sway_offset.x(),
359  velocity_target.x());
360  torso_trajectory.add_waypoint(X, step_period, velocity_target.x() * half_step_period, 0);
361 
362  // Y position trajectory
363  torso_trajectory.add_waypoint(Y, 0, Hpt_start.translation().y(), 0);
364  Scalar torso_offset_y = left_foot_is_planted ? -torso_sway_offset.y() : torso_sway_offset.y();
365  torso_trajectory.add_waypoint(Y, torso_sway_ratio * step_period, torso_offset_y, velocity_target.y());
366  torso_trajectory.add_waypoint(Y,
367  step_period,
368  get_foot_width_offset() / 2 + velocity_target.y() * half_step_period,
369  0);
370 
371  // Z position trajectory
372  torso_trajectory.add_waypoint(Z, 0, Hpt_start.translation().z(), 0);
373  torso_trajectory.add_waypoint(Z, torso_sway_ratio * step_period, torso_height + torso_sway_offset.z(), 0);
374  torso_trajectory.add_waypoint(Z, step_period, torso_height, 0);
375 
376  // Roll trajectory
377  torso_trajectory.add_waypoint(ROLL, 0, 0, 0);
378  torso_trajectory.add_waypoint(ROLL, torso_sway_ratio * step_period, 0, 0);
379  torso_trajectory.add_waypoint(ROLL, step_period, 0, 0);
380 
381  // Pitch trajectory
382  torso_trajectory.add_waypoint(PITCH, 0, torso_pitch, 0);
383  torso_trajectory.add_waypoint(PITCH, torso_sway_ratio * step_period, torso_pitch, 0);
384  torso_trajectory.add_waypoint(PITCH, step_period, torso_pitch, 0);
385 
386  // Yaw trajectory
387  torso_trajectory.add_waypoint(YAW, 0, MatrixToEulerIntrinsic(Hpt_start.linear()).z(), 0);
388  torso_trajectory.add_waypoint(YAW,
389  torso_sway_ratio * step_period,
390  velocity_target.z() * torso_sway_ratio * step_period,
391  0);
392  torso_trajectory.add_waypoint(YAW, step_period, velocity_target.z() * step_period, 0);
393  }

395  /**
396  * @brief Switch planted foot.
397  * @details This function switches the planted foot and updates the start torso and start swing foot poses.
398  */
399  void switch_planted_foot() {
400  // Transform planted foot to swing foot frame at end of step
401  Hps_start = get_swing_foot_pose(step_period).inverse();
402 
403  // Transform torso to end torso frame at end of step, in the new planted foot frame
404  Hpt_start = Hps_start * get_torso_pose(step_period);
405 
406  // Switch planted foot indicator
407  left_foot_is_planted = !left_foot_is_planted;
408 
409  // Reset time
410  t = 0;
411  }
412 
413  /**
414  * @brief Update the time.
415  * @param dt Time step.
416  */
417  void update_time(const Scalar& dt) {
418  // Check for negative time step
419  if (dt <= 0.0f) {
420  NUClear::log<NUClear::WARN>("dt <= 0.0f");
421  return;
422  }
423 
424  // Check for too long dt
425  if (dt > step_period) {
426  NUClear::log<NUClear::WARN>("dt > params.step_period");
427  return;
428  }
429 
430  // Update the phase
431  t += dt;
432 
433  // Clamp time to step period
434  if (t >= step_period) {
435  t = step_period;
436  }
437  }
438  };
